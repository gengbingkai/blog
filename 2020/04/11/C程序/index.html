#include "stdio.h"
#include "windows.h"
void s1(char arr[],int len){//项目一 
    for(int a=0;a<len;a++){
        printf("请输入第%d个元素：",a+1); //循环输入 
        scanf("%d",&arr[a]); 
    }
}
void s2(char arr[],int len){//项目二 
    int w=0;//定义总额 
    for(int a=0;a<len;a++){//循环相加 
        w=w+arr[a];
    }
    printf("数组平均值：%d",w/len); //输出平均值 
}
void s3(char arr[],int len){//项目三 
    int x;//定义变量 
    for(int a=0;a<len;a++){//双循环实现交换 
        for(int b=0;b<b-len;b++){
            if(arr[a]<arr[a+1]){
                x=arr[a];
                arr[a]=arr[a+1];
                arr[a+1]=x;
            }
        }
            printf("%d  ",arr[a]); //循环排序输出 

    }
}
void sss(){//定义菜单列表 

    printf("\t\t欢迎来闯关\n"); 
    printf("\t1：输入5个整数\n"); 
    printf("\t2：求平均值\n"); 
    printf("\t3：排序--降序\n"); 
    printf("\t0：安全退出！\n"); 
    while(1){
        int a;
        char arr[5];//死循环输入编号 
        printf("\n请输入功能编号："); 
        scanf("%d",&a); 
        switch(a){
            case 1:{
                s1(arr,5);//调用项目一 
                break;
            }
            case 2:{
                s2(arr,5);//调用项目二 
                break;
            }
            case 3:{ 
            s3(arr,5);//调用项目三 
                break;
            }
            case 0:{
                exit(0);//退出死循环 
                break;
            }
        }
    }
}
main(){
sss();//调用菜单列表
